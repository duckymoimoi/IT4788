// Khoi tao:
type AuthService struct {
    repo *repository.UserRepo
}
func NewAuthService(repo *repository.UserRepo) *AuthService

// Cac ham can viet:
Login(phone, password string) (*LoginResult, error)
    // Goi repo.FindByPhoneWithStaff -> bcrypt.Compare -> response.GenerateToken
    // Tra ve token, user_id, role

Signup(phone, password, fullName string) (*SignupResult, error)
    // Goi repo.FindByPhone kiem tra trung -> bcrypt.Hash -> repo.Create
    // Tao OTP, goi repo.CreateOTP
    // Tao setting mac dinh, goi repo.CreateSetting
    // Tra ve otp_code (debug, khong gui SMS)

VerifyOTP(phone, code, otpType string) (*VerifyResult, error)
    // Goi repo.FindValidOTP -> so sanh code -> repo.MarkOTPUsed
    // Neu otpType = "signup", doi user status = active
    // Tra ve token (user duoc login luon sau verify)

Logout(userID uint64, fcmToken string) error
    // Goi repo.DeactivateFCMToken

ForgotPassword(phone string) (*ForgotResult, error)
    // Goi repo.FindByPhone kiem tra ton tai -> tao OTP reset_password
    // Goi repo.CreateOTP, tra ve otp_code (debug)

ResetPassword(phone, code, newPassword string) error
    // Goi repo.FindValidOTP verify -> repo.MarkOTPUsed
    // bcrypt.Hash password moi -> repo.UpdatePassword

ChangePassword(userID uint64, oldPass, newPass string) error
    // Goi repo.FindByID -> bcrypt.Compare old -> bcrypt.Hash new
    // repo.UpdatePassword

Buoc 8 — handler/routes.go
Nhiem vu: dang ky tat ca route, phan nhom public/private.

func RegisterRoutes(router *gin.Engine, db *gorm.DB) {
    // Khoi tao cac tang
    userRepo    := repository.NewUserRepo(db)
    authService := service.NewAuthService(userRepo)
    userService := service.NewUserService(userRepo)
    authHandler := NewAuthHandler(authService)
    userHandler := NewUserHandler(userService)
    sysHandler  := NewSysHandler(userService)

    // Public routes - khong can token
    public := router.Group("/api")
    public.POST("/auth/login",           authHandler.Login)
    public.POST("/auth/signup",          authHandler.Signup)
    public.POST("/auth/verify_otp",      authHandler.VerifyOTP)
    public.POST("/auth/forgot_password", authHandler.ForgotPassword)
    public.POST("/auth/reset_password",  authHandler.ResetPassword)
    public.GET("/sys/check_version",     sysHandler.CheckVersion)

    // Private routes - can token
    private := router.Group("/api")
    private.Use(middleware.Auth())
    private.POST("/auth/logout",          authHandler.Logout)
    private.POST("/auth/change_password", authHandler.ChangePassword)
    private.GET("/user/get_profile",      userHandler.GetProfile)
    private.POST("/user/set_profile",     userHandler.SetProfile)
    private.POST("/user/set_devtoken",    userHandler.SetDevToken)
    private.GET("/user/get_settings",     userHandler.GetSettings)
    private.POST("/user/set_settings",    userHandler.SetSettings)
    private.DELETE("/user/delete_account", userHandler.DeleteAccount)
}
